状态模式

发布于：2020-01-07 00:00   标签分类： 设计模式

拆分状态跳转逻辑，每个状态在自己的内部，跳转到下一个状态。需要一个上下文类(Context)，状态基类(state)，具体状态类


<?php
class Context
{
    private $state;

    public function __construct($state)
    {
        $this->state = $state;
    }

    public function setState($state) {
        $this->state = $state;
    }

    public function getState() {
        return $this->state;
    }

    public function request() {
        $this->state->Handle($this);
    }
}

interface State
{
    public function Handle($context);
}

class StateA implements State
{
    public function Handle($context) {
        echo "state a run " . PHP_EOL;
        $context->setState(new StateB());
        $context->request();
    }
}

class StateB implements State
{
    public function Handle($context) {
        echo "state b run " . PHP_EOL;
        $context->setState(new StateC());
        $context->request();
    }
}

class StateC implements State
{
    public function Handle($context) {
        echo "state c run " . PHP_EOL;
    }
}

$context = new Context(new StateA());
$context->request();

输出：
state a run
state b run
state c run
