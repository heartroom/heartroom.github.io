访问者模式

发布于：2020-01-15 23:10   标签分类： 设计模式

需要对一个对象结构中的对象进行很多不同的并且不相关的操作，而需要避免让这些操作“污染”这些对象的类时，可以采用访问者模式，被访问者把自己的数据提供给访问者(visitor)。在被访问的类里面添加一个对外提供接待访问者的接口。把处理从数据结构分离出来。

<?php
interface Visitor
{
    public function visitElementA($elementA);
    public function visitElementB($elementB);
}

abstract class Element
{
    abstract public function accept($visitor);
}

class ElementA extends Element
{
    public $name;
    public function __construct($name)
    {
        $this->name = $name;
    }

    /**
     * @param $visitor Visitor
     */
    public function accept($visitor)
    {
        $visitor->visitElementA($this);
    }
}

class ElementB extends Element
{
    /**
     * @param $visitor Visitor
     */
    public function accept($visitor)
    {
        $visitor->visitElementB($this);
    }
}

class Visitor1 implements Visitor
{
    public function visitElementA($elementA)
    {
        echo 'Visitor1 访问了类型'.$elementA->name.PHP_EOL;
    }

    public function visitElementB($elementB)
    {
        echo 'Visitor1 访问了类型'.$elementB->name.PHP_EOL;
    }
}

class Visitor2 implements Visitor
{
    public function visitElementA($elementA)
    {
        echo 'Visitor2 访问了类型'.$elementA->name.PHP_EOL;
    }

    public function visitElementB($elementB)
    {
        echo 'Visitor2 访问了类型'.$elementB->name.PHP_EOL;
    }
}

class ObjectStructure
{
    private $elementList = [];
    public function add($element)
    {
        $this->elementList[] = $element;
    }

    public function accept($visitor)
    {
        foreach ($this->elementList as $item) {
            $item->accept($visitor);
        }
    }
}

$visitor1 = new Visitor1();
$visitor2 = new Visitor2();

$structure = new ObjectStructure();
$structure->add(new ElementA('aaa'));
$structure->add(new ElementA('aaa2'));
$structure->add(new ElementA('bbb2'));
$structure->add(new ElementA('ccc2'));

$structure->accept($visitor1);
$structure->accept($visitor2);
