享元模式

发布于：2020-01-14 00:00   标签分类： 设计模式

共享对象，对象池技术，缓存一类对象。

参考https://www.cnblogs.com/adamjwh/p/9070107.html

<?php
abstract class Flyweight
{
    protected $type;
    protected $innerData;
    public function __construct($type)
    {
        $this->type = $type;
    }

    abstract public function run($msg);
}

class ConcreteFlyweight extends Flyweight
{
    public function run($msg)
    {
        echo 'msg is ' . $msg;
    }
}

class FlyweightFactory
{
    private $pool = [];

    public function getFlyweight($type)
    {
        if (!$this->pool[$type]) {
            $this->pool[$type] = new ConcreteFlyweight($type);
            echo '创建新的对象'.$type.PHP_EOL;
        }
        return $this->pool[$type];
    }
}
$data = 10;
$flyweightFactory = new FlyweightFactory();
$flyweightA = $flyweightFactory->getFlyweight('A');
$flyweightA->run($data++);
$flyweightA = $flyweightFactory->getFlyweight('B');
$flyweightA->run($data++);
$flyweightA = $flyweightFactory->getFlyweight('A');
$flyweightA->run($data++);
