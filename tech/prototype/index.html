原型模式

发布于：2020-01-15 09:10   标签分类： 设计模式

根据一个对象创建一个完全相同的对象。clone。

<?php
class Prototype
{
    public function getClone()
    {
        return clone $this;
    }
}
class ConcretePrototype extends Prototype
{
    public $name;

    public function __construct($name)
    {
        $this->name = $name;
    }
}

$p1 = new ConcretePrototype('xxx');
$p2 = $p1->getClone();
echo $p2->name;
