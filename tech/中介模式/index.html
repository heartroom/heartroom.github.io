中介模式

发布于：2020-01-11 00:00   标签分类： 设计模式

中介者模式就是为了解决多个对象之间的复杂依赖关系。把依赖关系进行抽象，在中介对象里进行多个对象之间的复杂逻辑调用，降低了类之间的耦合，是迪米特法则(最少知道原则)的应用。MVC模式就是中介者模式，C是中介者。


<?php
abstract class AbstractMediator
{
    abstract public function run($action, $param);
}

class Role
{
    protected $mediator;

    public function setMediator($mediator)
    {
        $this->mediator = $mediator;
    }
}

class Programmer extends Role
{
    public function doWork()
    {
        $codeLine = mt_rand(10, 100);
        echo '程序员在努力工作中....' . PHP_EOL;
        echo '写了'.$codeLine.'行代码' . PHP_EOL;
        // 请求发工资
        $this->mediator->run('salary_request', $codeLine);
    }

    public function getMoney($money)
    {
        echo '收到工资' . $money. PHP_EOL;
    }
}

class Boss extends Role
{
    public function checkWork($codeLine)
    {
        echo '老板检查工作' . PHP_EOL;

        $money = 0;
        if ($codeLine>=90) {
            $money = 1000;
        } else if ($codeLine>=60) {
            $money = 500;
        } else if ($codeLine>=40) {
            $money = 100;
        }

        // 财务发工资
        $this->mediator->run('call_finance', $money);
    }
}

class Finance extends Role
{
    public function giveMoney($money)
    {
        echo '财务记账发工资' . $money;
        $money -= 100;// 扣保险金
        $this->mediator->run('pay_salary', $money);
    }
}

class Mediator extends AbstractMediator
{
    private $programmer;
    private $boss;
    private $finance;

    public function __construct()
    {
        $this->programmer = new Programmer();
        $this->programmer->setMediator($this);
        $this->boss = new Boss();
        $this->boss->setMediator($this);
        $this->finance = new Finance();
        $this->finance->setMediator($this);
    }

    public function run($action, $param)
    {
        switch ($action) {
            case 'salary_request':
                $this->boss->checkWork($param);
                break;
            case 'call_finance':
                $this->finance->giveMoney($param);
                break;
            case 'pay_salary':
                $this->programmer->getMoney($param);
                break;
        }
    }

    public function work()
    {
        $this->programmer->doWork();
    }
}

$mediator = new Mediator();
$mediator->work();
