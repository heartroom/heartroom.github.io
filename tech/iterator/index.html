迭代器模式

发布于：2020-01-15 23:10   标签分类： 设计模式

按顺序访问一个集合中的元素，不暴露元素的属性。这个模式已经C++，java里实现了。

<?php
interface MyIterator
{
    function next();
    function isEnd();
    function current();
}

class Iterator1 implements MyIterator
{
    private $list;
    private $index;

    public function __construct($list)
    {
        $this->list = $list;
        $this->index = 0;
    }

    public function next()
    {
        if ($this->isEnd()) {
            return null;
        } else {
            return $this->list[$this->index++];
        }
    }

    public function current()
    {
        return $this->list[$this->index];
    }

    public function remove()
    {
        array_splice($this->list, $this->index, 1);
    }

    public function isEnd()
    {
        return $this->index == count($this->list)-1;
    }
}

interface Aggregate
{
    function add($element);
    function remove($element);
    function iterator();
}

class ConcreteAggregate implements Aggregate
{
    private $list;

    public function __construct()
    {
        $this->list = [];
    }

    public function add($element)
    {
        $this->list[] = $element;
    }

    public function remove($element)
    {
        $key = array_search($element, $this->list);
        array_splice($this->list, $key, 1);
    }

    public function iterator()
    {
        return new Iterator1($this->list);
    }
}

$aggregate = new ConcreteAggregate();
$aggregate->add("aaa");
$aggregate->add("ddd");
$aggregate->add("bbb");
$aggregate->add("ccc");
$aggregate->add("ddd");

$iterator = $aggregate->iterator();

do{
    echo $iterator->current() . PHP_EOL;
} while($iterator->next());
