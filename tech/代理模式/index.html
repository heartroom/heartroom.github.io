代理模式

发布于：2020-01-11 00:00   标签分类： 设计模式

代理模式就是在调用真实对象方法之间增加了一层。代理和真实对象实现相同的接口，代理方法可以在真实对象方法之前或之后做一些特殊操作。AOP就是代理模式实现的。


<?php
interface Persion
{
    public function work($job);
}

class Worker implements Persion
{
    public function work($job)
    {
        echo '码农拼命的工作'.$job.PHP_EOL;
    }
}

class Boss implements Persion
{
    /** @var Worker */
    protected $worker;

    public function __construct($worker)
    {
        $this->worker = $worker;
    }

    public function work($job)
    {
        echo '收到外包工作'.PHP_EOL;
        $this->worker->work($job);
        echo '赚了100，给工人50'.PHP_EOL;
    }
}


$boss = new Boss(new Worker());
$boss->work('做个APP');
